CREATE TABLE IF NOT EXISTS clientes ( id BIGSERIAL PRIMARY KEY, nombre TEXT NOT NULL, empresa TEXT NOT NULL, email TEXT DEFAULT '', telefono TEXT DEFAULT '', activo BOOLEAN DEFAULT true, created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS proveedores ( id BIGSERIAL PRIMARY KEY, nombre TEXT NOT NULL, contacto TEXT DEFAULT '', email TEXT DEFAULT '', telefono TEXT DEFAULT '', categoria TEXT DEFAULT 'Materiales', activo BOOLEAN DEFAULT true, created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS proyectos ( id BIGSERIAL PRIMARY KEY, nombre TEXT NOT NULL, cliente_id BIGINT REFERENCES clientes(id) ON DELETE SET NULL, fecha_inicio DATE, fecha_fin DATE, estatus TEXT DEFAULT 'Planeación' CHECK (estatus IN ('Planeación','En Progreso','Completado','Pausado')), presupuesto NUMERIC(15,2) DEFAULT 0, avance INTEGER DEFAULT 0 CHECK (avance BETWEEN 0 AND 100), notas TEXT DEFAULT '', created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS requisiciones ( id BIGSERIAL PRIMARY KEY, proyecto_id BIGINT NOT NULL REFERENCES proyectos(id) ON DELETE CASCADE, descripcion TEXT NOT NULL, monto NUMERIC(15,2) DEFAULT 0, proveedor_id BIGINT REFERENCES proveedores(id) ON DELETE SET NULL, estatus TEXT DEFAULT 'Pendiente' CHECK (estatus IN ('Pendiente','Aprobada','Pagada','Rechazada')), fecha DATE DEFAULT CURRENT_DATE, notas TEXT DEFAULT '', created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE OR REPLACE FUNCTION fn_login(p_email TEXT, p_password TEXT) RETURNS TABLE(id BIGINT,nombre TEXT,email TEXT,departamento TEXT,rol TEXT, perm_empleados BOOLEAN,perm_clientes BOOLEAN,perm_proveedores BOOLEAN, perm_proyectos BOOLEAN,perm_ventas BOOLEAN) LANGUAGE plpgsql SECURITY DEFINER AS $$ BEGIN UPDATE empleados SET ultimo_acceso=NOW() WHERE email=p_email AND password=crypt(p_password,password) AND activo=true; RETURN QUERY SELECT e.id,e.nombre,e.email,e.departamento,e.rol, e.perm_empleados,e.perm_clientes,e.perm_proveedores,e.perm_proyectos,e.perm_ventas FROM empleados e WHERE e.email=p_email AND e.password=crypt(p_password,e.password) AND e.activo=true; END; $$; CREATE OR REPLACE FUNCTION fn_crear_empleado( p_nombre TEXT,p_email TEXT,p_password TEXT,p_departamento TEXT,p_rol TEXT, p_perm_empleados BOOLEAN,p_perm_clientes BOOLEAN,p_perm_proveedores BOOLEAN, p_perm_proyectos BOOLEAN,p_perm_ventas BOOLEAN) RETURNS BIGINT LANGUAGE plpgsql SECURITY DEFINER AS $$ DECLARE v_id BIGINT; BEGIN INSERT INTO empleados(nombre,email,password,departamento,rol,activo, perm_empleados,perm_clientes,perm_proveedores,perm_proyectos,perm_ventas) VALUES(p_nombre,p_email,crypt(p_password,gen_salt('bf')),p_departamento,p_rol,true, p_perm_empleados,p_perm_clientes,p_perm_proveedores,p_perm_proyectos,p_perm_ventas) RETURNING id INTO v_id; RETURN v_id; END; $$; CREATE OR REPLACE FUNCTION fn_cambiar_password(p_id BIGINT, p_nueva TEXT) RETURNS VOID LANGUAGE plpgsql SECURITY DEFINER AS $$ BEGIN UPDATE empleados SET password=crypt(p_nueva,gen_salt('bf')) WHERE id=p_id; END; $$; ALTER TABLE empleados DISABLE ROW LEVEL SECURITY; ALTER TABLE clientes DISABLE ROW LEVEL SECURITY; ALTER TABLE proveedores DISABLE ROW LEVEL SECURITY; ALTER TABLE proyectos DISABLE ROW LEVEL SECURITY;
